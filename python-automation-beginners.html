1. File Organizer

import os
import shutil

def organize_files(folder):
    for file in os.listdir(folder):
        ext = os.path.splitext(file)[1]
        folder_path = os.path.join(folder, ext[1:])
        os.makedirs(folder_path, exist_ok=True)
        shutil.move(os.path.join(folder, file), folder_path)

2. Email Sender

import smtplib

def send_email(to, subject, body):
    server = smtplib.SMTP('smtp.gmail.com', 587)
    server.starttls()
    server.login('your_email@gmail.com', 'password')
    server.sendmail('your_email@gmail.com', to, f'Subject: {subject}\n\n{body}')
    server.quit()

3. Backup Script

import shutil
import datetime

def backup(source, dest):
    timestamp = datetime.datetime.now().strftime('%Y%m%d_%H%M%S')
    shutil.copytree(source, f'{dest}/backup_{timestamp}')
